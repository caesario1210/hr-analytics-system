Schema

CREATE TABLE employee (
  employee_id INT PK, full_name VARCHAR(150),
  salary NUMERIC(12,2), status VARCHAR(20),
  department_id INT FK, position_id INT FK
);
-- 8 tables total (see create_table.sql)

Try This Query

SELECT d.department_name,
  ROUND(AVG(e.salary),2) AS avg_salary
FROM employee e
JOIN department d ON e.department_id = d.department_id
WHERE e.status = 'Active'
GROUP BY d.department_name
ORDER BY avg_salary DESC;

Sample SQL Queries

Showcasing window functions, subqueries, and aggregation techniques.

Salary Rank by Department

SELECT full_name, department_name, salary,
   RANK() OVER (PARTITION BY department_name ORDER BY salary DESC) AS salary_rank
FROM employee e
JOIN department d ON e.department_id = d.department_id;

Turnover Rate

SELECT
  ROUND(
    100.0 * SUM(CASE WHEN status <> 'Active' THEN 1 ELSE 0 END) / COUNT(*), 2
  ) AS turnover_rate_percent
FROM employee;

Recruitment Funnel

SELECT
  status, COUNT(*) AS count
FROM recruitment
GROUP BY status
ORDER BY count DESC;

Avg KPI per Period

SELECT period, AVG(kpi_score) AS avg_kpi
FROM performance
GROUP BY period
ORDER BY period;
